#include <iostream>
using namespace std;

void printArray(int arr[],int size){
    for(int i=0;i<size;i++){
        cout<<arr[i]<<" ";
    }
    cout<<endl;
}

void merge(int arr[],int l,int m,int r){
    int i,j,k;
    int n1 = m - l + 1;
    int n2 = r - m;
    cout<<"L:"<<l<<" M:"<<m<<" R:"<<r<<" N1:"<<n1<<" N2:"<<n2<<endl;
    int L[n1],R[n2];
    
    for(i=0;i<n1;i++)
        L[i]=arr[l+i];
    for(j=0;j<n2;j++)
        R[j]=arr[m + 1 + j];
        
    cout<<"=============================Array L====================="<<endl;
    printArray(L,n1);
    cout<<"=============================Array R====================="<<endl;
    printArray(R,n2);
    cout<<"========================================================="<<endl;
    cout<<endl;
        
    i=0;
    j=0;
    k=0;
    
    while(i<n1 && j<n2){
        if(L[i]<=R[j]){
            arr[k]=L[i];
            i++;
        }
        else{
            arr[k]=R[j];
            j++;
        }
        k++;
    }
    
    while(i<n1){
        arr[k]=L[i];
        i++;
        k++;
    }
    
    while(j<n2){
        arr[k]=R[j];
        j++;
        k++;
    }
}


void mergeSort(int arr[],int l,int r){
    if(l<r){
        int m = l + (r-l)/2;
        cout<<"L:"<<l<<" R:"<<r<<" M:"<<m<<endl;
        mergeSort(arr,l,m);
        cout<<"L:"<<l<<" R:"<<r<<" M:"<<m<<endl;
        mergeSort(arr,m+1,r);
        cout<<"L:"<<l<<" R:"<<r<<" M:"<<m<<endl;
        
        merge(arr,l,m,r);
    }
}

int main()
{
    cout<<"Hello World"<<endl;
    int arr[] = {1,2,3,4,5,6,7,8,9,8,7,6,5,4,3,2,1};
    int arrSize=sizeof(arr)/sizeof(arr[0]);
    
    cout<<"===============Before Sorting======================"<<endl;
    printArray(arr,arrSize);
    
    cout<<"Array Size:"<<arrSize<<endl;
    mergeSort(arr,0,arrSize-1);
    
    cout<<"===============After Sorting======================"<<endl;
    printArray(arr,arrSize);
    return 0;
}
